def timeLong(n: Long): Double = {
  timer.reset
  var i = 0L

  timer.elapsedSeconds

  timer.reset
  var i = 0
  while (i < n) { i += 1 }
  timer.elapsedSeconds

  if (n <= Int.MaxValue) show("Int  " + n, timeInt(n.toInt))
