  def nextRandomPos(): Pos = 
    import scala.util.Random.nextInt

    val y = nextInt(BlockWindow.windowSize._2 - 7) + 7
    (x, y)

    if math.random() < Worm.teleportProbability then

        pos == notHere

end Worm
